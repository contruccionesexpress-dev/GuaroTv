    <!-- SCRIPT -->
    <script>
        let activeRole = 'viewer';
        let currentActiveChannel = 1;
        let currentFilter = 'ALL';
        let mediaRecorders = { 1: null, 2: null, 3: null };
        let recordedChunks = { 1: [], 2: [], 3: [] };
        let streamSources = { 1: null, 2: null, 3: null };

        let channelAdjustments = {
            1: { cropTop: 30, cropBottom: 0, zoom: 110, offsetX: 0 },
            2: { cropTop: 0, cropBottom: 0, zoom: 100, offsetX: 0 },
            3: { cropTop: 0, cropBottom: 0, zoom: 100, offsetX: 0 }
        };

        let listGacetas = [
            { id: 1, title: "Gaceta Hípica Oficial", track: "La Rinconada", country: "VEN", date: "2026-06-14", link: "https://www.desafiohipico.com/retrospectos" },
            { id: 2, title: "Línea del Éxito", track: "Valencia", country: "VEN", date: "2026-06-13", link: "https://www.desafiohipico.com/retrospectos" },
            { id: 3, title: "Daily Racing Form", track: "Gulfstream Park", country: "USA", date: "2026-06-14", link: "https://grtv.us/" }
        ];

        window.onload = function() {
            const urlParams = new URLSearchParams(window.location.search);
            const isGuaroRoom = urlParams.get('sala') === 'elguaro' || window.location.hash === '#elguaro';

            if (isGuaroRoom) {
                changeRole('viewer');
                document.getElementById('roleSelectorContainer').style.display = 'none';
                document.getElementById('guaroWelcomeHeader').classList.remove('hidden');
                triggerNotification("Conexión VIP", "Bienvenido a la señal premium de El Guaro.", "👑");
            } else {
                changeRole('viewer');
            }

            // Capturar la URL real de publicación (Replit la inyecta automáticamente aquí)
            const currentUrl = window.location.href.split('?')[0].split('#')[0];
            document.getElementById('realLinkDisplay').innerText = `${currentUrl}?sala=elguaro`;

            const stored = localStorage.getItem('elguaro_gacetas');
            if (stored) listGacetas = JSON.parse(stored);
            
            renderGacetas();
            applyAllChannelAdjustments();
            document.getElementById('adminPdfDate').value = new Date().toISOString().split('T')[0];
        };

        function copyGeneratedLink(type) {
            const textToCopy = document.getElementById('realLinkDisplay').innerText;
            const tempInput = document.createElement("input");
            tempInput.value = textToCopy;
            document.body.appendChild(tempInput);
            tempInput.select();
            
            try {
                document.execCommand("copy");
                triggerNotification("Enlace Copiado", "Listo para pegar en tu grupo hípico de WhatsApp.", "📋");
            } catch (err) {
                triggerNotification("Error al copiar", "Selecciónalo manualmente.", "❌", true);
            }
            document.body.removeChild(tempInput);
        }

        function triggerNotification(title, message, icon = "🛡️", isAlert = false) {
            const toast = document.getElementById('toastNotification');
            document.getElementById('toastTitle').innerText = title;
            document.getElementById('toastMessage').innerText = message;
            document.getElementById('toastIcon').innerText = icon;

            if (isAlert) {
                toast.classList.remove('bg-emerald-700');
                toast.classList.add('bg-red-700');
            } else {
                toast.classList.remove('bg-red-700');
                toast.classList.add('bg-emerald-700');
            }

            toast.classList.remove('translate-x-full');
            setTimeout(() => toast.classList.add('translate-x-full'), 3000);
        }

        function changeRole(role) {
            activeRole = role;
            const btnViewer = document.getElementById('btnRoleViewer');
            const btnAdmin = document.getElementById('btnRoleAdmin');
            const adminArea = document.getElementById('adminControlsArea');
            const adminButtons = document.querySelectorAll('.admin-only');

            if (role === 'admin') {
                btnAdmin.className = "px-4 py-2 rounded-lg text-xs font-bold uppercase transition-all flex items-center space-x-2 bg-amber-500 text-black border border-amber-400 font-black shadow-md";
                btnViewer.className = "px-4 py-2 rounded-lg text-xs font-bold uppercase transition-all flex items-center space-x-2 text-slate-400 hover:text-white";
                adminArea.classList.remove('hidden');
                adminButtons.forEach(btn => btn.classList.remove('hidden'));
                loadAdjustmentsForSelectedChannel();
            } else {
                btnViewer.className = "px-4 py-2 rounded-lg text-xs font-bold uppercase transition-all flex items-center space-x-2 bg-emerald-700 text-white border border-emerald-500 shadow-md";
                btnAdmin.className = "px-4 py-2 rounded-lg text-xs font-bold uppercase transition-all flex items-center space-x-2 text-slate-400 hover:text-white";
                adminArea.classList.add('hidden');
                adminButtons.forEach(btn => btn.classList.add('hidden'));
            }
        }

        function setActiveChannel(chan) {
            currentActiveChannel = chan;
            for (let i = 1; i <= 3; i++) {
                const btn = document.getElementById(`btnChan${i}`);
                const wrapper = document.getElementById(`wrapperChannel${i}`);
                if (i === chan) {
                    btn.className = "px-3 py-1.5 rounded-lg text-xs font-bold uppercase transition-all bg-emerald-700 text-white border border-emerald-500 flex items-center gap-1.5";
                    wrapper.classList.remove('hidden');
                } else {
                    btn.className = "px-3 py-1.5 rounded-lg text-xs font-bold uppercase transition-all bg-black/40 text-slate-300 border border-red-900/40 hover:border-amber-500 flex items-center gap-1.5";
                    wrapper.classList.add('hidden');
                }
            }
            document.getElementById('btnMultiview').className = "px-3 py-1.5 rounded-lg text-xs font-bold uppercase transition-all bg-black/40 text-slate-300 border border-red-900/40 hover:border-amber-500 flex items-center gap-1.5";
            document.getElementById('gridStreamsContainer').className = "grid grid-cols-1 gap-4";
        }

        function setMultiviewMode() {
            currentActiveChannel = 'multiview';
            for (let i = 1; i <= 3; i++) {
                document.getElementById(`btnChan${i}`).className = "px-3 py-1.5 rounded-lg text-xs font-bold uppercase transition-all bg-black/40 text-slate-300 border border-red-900/40 flex items-center gap-1.5";
                document.getElementById(`wrapperChannel${i}`).classList.remove('hidden');
            }
            document.getElementById('btnMultiview').className = "px-3 py-1.5 rounded-lg text-xs font-bold uppercase transition-all bg-amber-500 text-black font-black border border-amber-400 flex items-center gap-1.5";
            document.getElementById('gridStreamsContainer').className = "grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-4";
        }

        function loadAdjustmentsForSelectedChannel() {
            const chanId = document.getElementById('selectedChannelToAdjust').value;
            const config = channelAdjustments[chanId];
            document.getElementById('rngCropTop').value = config.cropTop;
            document.getElementById('rngZoom').value = config.zoom;
            updateLabelsAndUI();
        }

        function updateLabelsAndUI() {
            const cropTop = document.getElementById('rngCropTop').value;
            const zoom = document.getElementById('rngZoom').value;
            document.getElementById('lblCropTop').innerText = cropTop + "px";
            document.getElementById('lblZoom').innerText = zoom + "%";
        }

        function updateCropAdjustments() {
            const chanId = document.getElementById('selectedChannelToAdjust').value;
            const cropTop = parseInt(document.getElementById('rngCropTop').value);
            const zoom = parseInt(document.getElementById('rngZoom').value);
            channelAdjustments[chanId] = { cropTop, cropBottom: 0, zoom, offsetX: 0 };
            applySingleChannelAdjustments(chanId);
            updateLabelsAndUI();
        }

        function applySingleChannelAdjustments(chanId) {
            const config = channelAdjustments[chanId];
            const videoElement = document.getElementById(`videoElement${chanId}`);
            const viewport = document.getElementById(`viewport${chanId}`);
            if (!videoElement) return;
            viewport.style.clipPath = `inset(${config.cropTop}px 0px 0px 0px)`;
            videoElement.style.transform = `scale(${config.zoom / 100})`;
        }

        function applyAllChannelAdjustments() {
            for (let i = 1; i <= 3; i++) applySingleChannelAdjustments(i);
        }

        async function startSourceCapture(channelId) {
            try {
                const stream = await navigator.mediaDevices.getDisplayMedia({
                    video: { cursor: "always" },
                    audio: true
                });
                const video = document.getElementById(`videoElement${channelId}`);
                video.srcObject = stream;
                video.play();
                streamSources[channelId] = stream;
                triggerNotification("Señal Transmitida", "Tu app externa se está retransmitiendo con éxito.", "📲");
            } catch (err) {
                triggerNotification("Captura cancelada", "No seleccionaste pantalla.", "❌", true);
            }
        }

        function toggleRecording(channelId) {
            const btn = document.getElementById(`btnRecord${channelId}`);
            const statusIndicator = document.getElementById(`recordingStatus${channelId}`);

            if (mediaRecorders[channelId] && mediaRecorders[channelId].state === "recording") {
                mediaRecorders[channelId].stop();
                btn.className = "bg-red-700 hover:bg-red-600 text-white font-bold text-xs px-3 py-2 rounded-lg transition-all border border-red-500";
                btn.innerHTML = `<i class="fa-solid fa-circle-dot"></i> Grabar Carrera`;
                statusIndicator.classList.add('hidden');
                return;
            }

            let streamToRecord = streamSources[channelId];
            if (!streamToRecord) {
                const videoEl = document.getElementById(`videoElement${channelId}`);
                streamToRecord = videoEl.captureStream ? videoEl.captureStream() : videoEl.mozCaptureStream();
            }

            recordedChunks[channelId] = [];
            try {
                const recorder = new MediaRecorder(streamToRecord, { mimeType: 'video/webm; codecs=vp9' });
                recorder.ondataavailable = e => { if (e.data.size > 0) recordedChunks[channelId].push(e.data); };
                recorder.onstop = () => {
                    const blob = new Blob(recordedChunks[channelId], { type: 'video/webm' });
                    const url = URL.createObjectURL(blob);
                    const a = document.createElement('a');
                    a.href = url;
                    a.download = `ElGuaro_Carrera_${Date.now()}.webm`;
                    a.click();
                    triggerNotification("Grabación Lista", "El archivo de la carrera se descargó de forma local.", "💾");
                };
                recorder.start();
                mediaRecorders[channelId] = recorder;
                btn.className = "bg-neutral-800 text-amber-400 font-bold text-xs px-3 py-2 rounded-lg transition-all animate-pulse border border-amber-500";
                btn.innerHTML = `<i class="fa-solid fa-square"></i> Detener`;
                statusIndicator.classList.remove('hidden');
            } catch (err) {
                triggerNotification("Error de Códec", "No soportado en este navegador.", "❌", true);
            }
        }

        function renderGacetas() {
            const container = document.getElementById('gacetasContainer');
            container.innerHTML = '';
            const filtered = listGacetas.filter(p => currentFilter === 'ALL' || p.country === currentFilter);

            if (filtered.length === 0) {
                container.innerHTML = `<div class="text-center py-4 text-neutral-600 text-xs">No hay revistas disponibles.</div>`;
                return;
            }

            filtered.forEach(pdf => {
                const flag = pdf.country === 'VEN' ? '🇻🇪' : '🇺🇸';
                const div = document.createElement('div');
                div.className = "bg-black p-3 rounded-xl border border-red-950 flex items-center justify-between hover:border-amber-500/30 transition-all";
                div.innerHTML = `
                    <div class="flex items-center space-x-2">
                        <div class="text-red-500 text-xl px-1"><i class="fa-solid fa-file-pdf"></i></div>
                        <div>
                            <h4 class="text-xs font-bold text-slate-200">${pdf.title}</h4>
                            <p class="text-[10px] text-amber-500/80 font-medium">${pdf.track} ${flag} • 📅 ${pdf.date}</p>
                        </div>
                    </div>
                    <button onclick="startDownloadSequence(${pdf.id})" class="bg-emerald-700 hover:bg-emerald-600 border border-emerald-500 text-white px-2.5 py-1 rounded text-[10px] font-bold uppercase transition-all">
                        Bajar
                    </button>
                `;
                container.appendChild(div);
            });
        }

        function filterPdfs(country) {
            currentFilter = country;
            ['ALL', 'VEN', 'USA'].forEach(type => {
                document.getElementById(`btnFilter_${type}`).className = type === country ? 
                    "flex-1 py-1 rounded text-[10px] font-bold uppercase transition-all bg-red-700 text-white" : 
                    "flex-1 py-1 rounded text-[10px] font-bold uppercase transition-all text-slate-400 hover:text-white";
            });
            renderGacetas();
        }

        function uploadGaceta() {
            const title = document.getElementById('adminPdfTitle').value.trim();
            const track = document.getElementById('adminPdfTrack').value.trim();
            const country = document.getElementById('adminPdfCountry').value;
            const date = document.getElementById('adminPdfDate').value;
            const link = document.getElementById('adminPdfLink').value.trim();

            if (!title || !track || !date) {
                triggerNotification("Campos Vacíos", "Rellena los datos requeridos.", "❌", true);
                return;
            }

            listGacetas.unshift({ id: Date.now(), title, track, country, date, link: link || '#' });
            localStorage.setItem('elguaro_gacetas', JSON.stringify(listGacetas));
            
            document.getElementById('adminPdfTitle').value = '';
            document.getElementById('adminPdfTrack').value = '';
            document.getElementById('adminPdfLink').value = '';
            
            renderGacetas();
            triggerNotification("Documento Indexado", "La revista ya está disponible para el público.", "📁");
        }

        let downloadInterval = null;
        function startDownloadSequence(id) {
            const pdf = listGacetas.find(p => p.id === id);
            if (!pdf) return;

            const modal = document.getElementById('modalDownload');
            const actionBtn = document.getElementById('dlActionBtn');
            const bar = document.getElementById('dlBar');
            const percent = document.getElementById('dlPercent');

            document.getElementById('dlTitle').innerText = "Abriendo " + pdf.title;
            document.getElementById('dlSubtitle').innerText = pdf.track + " - Programa Oficial";
            actionBtn.href = pdf.link !== "#" ? pdf.link : "https://www.desafiohipico.com/retrospectos";
            actionBtn.classList.add('hidden');
            bar.style.width = '0%';
            percent.innerText = '0%';
            modal.classList.remove('hidden');

            let p = 0;
            if (downloadInterval) clearInterval(downloadInterval);
            downloadInterval = setInterval(() => {
                p += 20;
                if (p >= 100) {
                    p = 100;
                    clearInterval(downloadInterval);
                    actionBtn.classList.remove('hidden');
                }
                bar.style.width = p + '%';
                percent.innerText = p + '%';
            }, 100);
        }

        function closeDlModal() {
            if (downloadInterval) clearInterval(downloadInterval);
            document.getElementById('modalDownload').classList.add('hidden');
        }

        function postChatMessage(e) {
            e.preventDefault();
            const input = document.getElementById('chatInput');
            const val = input.value.trim();
            if (!val) return;
            const chat = document.getElementById('chatBox');
            const div = document.createElement('div');
            div.className = "flex items-start space-x-2";
            div.innerHTML = `<span class="bg-red-950 text-red-400 text-[9px] font-black px-1.5 py-0.5 rounded border border-red-500/20">TÚ</span> <strong class="text-amber-400">Espectador:</strong> <span class="text-slate-200">${val}</span>`;
            chat.appendChild(div);
            input.value = '';
            chat.scrollTop = chat.scrollHeight;
        }
